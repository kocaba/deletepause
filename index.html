if(!file) return; ui.ov.style.display = 'flex'; ui.msg.innerText = "Читання аудіо..."; // Створюємо аудіо-контекст примусово const AudioContext = window.AudioContext || window.webkitAudioContext; const aCtx = new AudioContext(); try { if (aCtx.state === 'suspended') await aCtx.resume(); const arrayBuf = await file.arrayBuffer(); // СУМІСНІСТЬ З SAFARI (iOS): Використовуємо колбеки замість Promise aCtx.decodeAudioData(arrayBuf, (decoded) => { aBuf = decoded.getChannelData(0);
